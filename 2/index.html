Code Implementation


import numpy as np
import scipy.signal
from skimage import io as skio

# Finite difference operators
D_y = np.array([[1, 0, -1]])
D_x = np.array([[1], [0], [-1]])

def read_in_img(img_title, grayscale=False):
    """Read in an image file"""
    imname = img_title
    im = skio.imread(imname, as_gray=grayscale)
    return im

def pad_img(img, kernel_height, kernel_width):
    """Add zero padding to image based on kernel size"""
    pad_height = kernel_height // 2
    pad_width = kernel_width // 2
    
    padded_img = np.pad(img, ((pad_height, pad_height), (pad_width, pad_width)))
    return padded_img

def four_loop_convolve(image, kernel):
    """Convolution implementation using four nested loops"""
    image = pad_img(image, kernel.shape[0], kernel.shape[1])
    img_width = image.shape[1]
    img_height = image.shape[0]
    kernel_width = kernel.shape[1]
    kernel_height = kernel.shape[0]
    result_height = img_height - kernel_height + 1
    result_width = img_width - kernel_width + 1
    result = np.zeros((result_height, result_width))

    for img_row in range(result_height):
        for img_col in range(result_width):
            res = 0
            for kernel_row in range(kernel_height):
                for kernel_col in range(kernel_width):
                    flipped_kernel_row = kernel_height - 1 - kernel_row
                    flipped_kernel_col = kernel_width - 1 - kernel_col
                    res += image[img_row + kernel_row, img_col + kernel_col] * kernel[flipped_kernel_row, flipped_kernel_col]
            result[img_row][img_col] = res 
    return result

def two_loop_convolve(image, kernel):
    """Convolution implementation using two nested loops with vectorized operations"""
    image = pad_img(image, kernel.shape[0], kernel.shape[1])
    img_width = image.shape[1]
    img_height = image.shape[0]
    kernel_width = kernel.shape[1]
    kernel_height = kernel.shape[0]
    result_height = img_height - kernel_height + 1
    result_width = img_width - kernel_width + 1
    result = np.zeros((result_height, result_width))
    
    flipped_kernel = np.flip(kernel)
    for res_row in range(result_height):
        for res_col in range(result_width):
            img_start_row = res_row
            img_end_row = img_start_row + kernel_height
            img_start_col = res_col
            img_end_col = img_start_col + kernel_width
            img_section = image[img_start_row:img_end_row, img_start_col:img_end_col]
            result[res_row][res_col] = np.sum(img_section * flipped_kernel)
    return result

def question_1_1():
    """Test convolution implementations with box filter and finite difference operators"""
    box_filter = np.ones((9,9))
    img = read_in_img("1.1.jpeg", grayscale=True)
    
    # Box filter results
    box_four_loop_convolve_res = four_loop_convolve(img, box_filter)
    box_two_loop_convolve_res = two_loop_convolve(img, box_filter)
    box_built_in_convolve_res = scipy.signal.convolve2d(img, box_filter, mode='same')

    # D_x filter results
    D_x_four_loop_convolve_res = four_loop_convolve(img, D_x)
    D_x_two_loop_convolve_res = two_loop_convolve(img, D_x)
    D_x_built_in_convolve_res = scipy.signal.convolve2d(img, D_x, mode='same')

    # D_y filter results
    D_y_four_loop_convolve_res = four_loop_convolve(img, D_y)
    D_y_two_loop_convolve_res = two_loop_convolve(img, D_y)
    D_y_built_in_convolve_res = scipy.signal.convolve2d(img, D_y, mode='same')
    
    return
